name: Build Tsmart Pro EXE (Win11 Stable)
on: [push, workflow_dispatch]
jobs:
  build:
    runs-on: windows-latest
    steps:
    - uses: actions/checkout@v4
    - uses: actions/setup-python@v5
      with: {python-version: '3.10'}
    - name: Install Dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
    - name: Build Standalone EXE with Libusb Features
      run: |
        # Building with libusb, libusb-exp, and serial features for Win11 stability
        pyinstaller --onefile --windowed \
          --add-data "mtkclient;mtkclient" \
          --add-data "unisoc;unisoc" \
          --add-data "penumbra;penumbra" \
          --add-data "bin;bin" \
          --name "Tsmart_Pro_Tool_v2.1" \
          --hidden-import="usb.backend.libusb1" \
          --hidden-import="serial" \
          main.py
    - name: Upload Artifact
      uses: actions/upload-artifact@v4
      with:
        name: Tsmart_Pro_Win11_Stable
        path: dist/Tsmart_Pro_Tool_v2.1.exe
